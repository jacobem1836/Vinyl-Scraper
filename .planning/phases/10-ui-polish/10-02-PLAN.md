Replace the startScan function with:
```javascript
function startScan(itemId) {
  // Disable all scan buttons
  document.querySelectorAll('[data-scan-btn]').forEach(function(btn) {
    btn.disabled = true;
  });

  var url = (itemId != null) ? '/api/scan/start?item_id=' + itemId : '/api/scan/start';
  fetch(url, { method: 'POST', headers: { 'X-API-Key': window.VINYL_API_KEY || '' } })
    .then(function(r) { return r.json(); })
    .then(function(data) {
      if (data.started) { showPanel(); startPolling(); }
      else {
        // Scan didn't start — re-enable buttons
        document.querySelectorAll('[data-scan-btn]').forEach(function(btn) {
          btn.disabled = false;
        });
      }
    })
    .catch(function(err) {
      console.error(err);
      // Error — re-enable buttons
      document.querySelectorAll('[data-scan-btn]').forEach(function(btn) {
        btn.disabled = false;
      });
    });
}
```

Also update `renderStatus` to re-enable buttons when scan completes. In the `else` branch of `renderStatus` (scan is done, around line 213-219), after `stopPolling()` add:
